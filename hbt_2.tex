\coordinate (S1) at (1.8, 5.5);

}

    \fill[starglow, opacity=\op] (S2) circle (\r);
}
\fill[white] (S2) circle (0.04);
\foreach \ang in {0,45,90,135}{
    \draw[white, line width=0.6pt, opacity=0.8]
        ($(S2)+(\ang:0.06)$) -- ($(S2)+(\ang:0.20)$);

\node[text=source2col, font=\small\bfseries, left=2pt] at (S2) {Source 2};

    ($(S1)+(0.1, 0)$) -- ($(TEL)+(0.0,-0.55)$);

    ($(S2)+(0.1, 0)$) -- ($(TEL)+(0.0,-0.55)$);

    ($(TEL)+(0.0, 0.7)$) -- ($(TEL)+(0.0,-0.7)$);

\node[text=labelcol, font=\small\bfseries, above=4pt] at ($(TEL)+(0.25, 0.7)$) {Telescope};

% 2. BEAM SPLITTER
% ════════════════════════════════════════════════════════

    ($(BS)+(-0.35,-0.35)$) -- ($(BS)+(0.35, 0.35)$);
\node[text=bscol, font=\scriptsize, below=10pt] at (BS) {BS};

% Output beam 1 (transmitted, horizontal)
\draw[arr, photoncol, opacity=0.8] ($(BS)+(0.35, 0.0)$) -- ($(BS)+(1.5, 0.0)$);

% Output beam 2 (reflected, vertical up)
\draw[arr, photoncol, opacity=0.8] ($(BS)+(0.0, 0.35)$) -- ($(BS)+(0.0, 1.5)$);
