]

     \textemdash\ $g^{(2)}(\tau)$};

% ════════════════════════════════════════════════════════
% 1. SOURCE + TELESCOPE (LEFT HALF)

\coordinate (S1) at (1.2, 5.5);
\coordinate (S2) at (1.2, 3.0);

% Glow halos S1

    \fill[starglow, opacity=\op] (S1) circle (\r);
}
\fill[white] (S1) circle (0.05);
\foreach \ang in {0,45,90,135}{
    \draw[white, line width=0.6pt, opacity=0.8]
        ($(S1)+(\ang:0.06)$) -- ($(S1)+(\ang:0.20)$);
}

% Glow halos S2
\foreach \r/\op in {0.25/15, 0.15/30, 0.08/60}{
    \fill[starglow, opacity=\op] (S2) circle (\r);
}
\fill[white] (S2) circle (0.05);
\foreach \ang in {0,45,90,135}{
    \draw[white, line width=0.6pt, opacity=0.8]
        ($(S2)+(\ang:0.06)$) -- ($(S2)+(\ang:0.20)$);
}
\node[text=source2col, font=\small\bfseries, left=2pt] at (S2) {Source 2};

\coordinate (TEL) at (5.0, 4.25);

\draw[photoncol, line width=0.9pt, opacity=0.7]
    ($(S1)+(0.1, 0)$) -- ($(TEL)+(0.0, 0.55)$);

    ($(S1)+(0.1, 0)$) -- ($(TEL)+(0.0,-0.55)$);

    ($(S2)+(0.1, 0)$) -- ($(TEL)+(0.0, 0.55)$);

    ($(S2)+(0.1, 0)$) -- ($(TEL)+(0.0,-0.55)$);

% Photon bullets
\fill[photoncol] (3.2, 4.6) circle (0.09);
\fill[photoncol] (3.7, 4.1) circle (0.09);

    ($(TEL)+(0.0, 0.7)$)

    ($(TEL)+(0.0, 0.7)$) -- ($(TEL)+(0.0,-0.7)$);

% ════════════════════════════════════════════════════════
% 2. DETECTOR + TIME SERIES
% ════════════════════════════════════════════════════════
\coordinate (DET) at (6.5, 4.25);

% Arrow from telescope to detector
\draw[arr, labelcol] ($(TEL)+(0.5, 0.0)$) -- ($(DET)+(-0.3, 0.0)$);

    ($(DET)+(-0.3,-0.4)$) rectangle ($(DET)+(0.3, 0.4)$);
\node[text=labelcol, font=\scriptsize] at (DET) {Det};

% Time series of photon clicks
\coordinate (TS) at (7.5, 4.25);
\draw[arr, axiscol] ($(TS)+(0, 0)$) -- ($(TS)+(4.5, 0)$);
\node[text=axiscol, font=\scriptsize] at ($(TS)+(4.7, 0)$) {$t$};

% Photon arrival tick marks
\foreach \x in {0.4, 1.1, 1.5, 2.3, 2.8, 3.5, 4.0}{
    \draw[photoncol, line width=1.5pt]
        ($(TS)+(\x, 0)$) -- ($(TS)+(\x, 0.4)$);
}

% Δτ label between two consecutive clicks (1.5 and 2.3)
\draw[white, line width=0.6pt, <->]
    ($(TS)+(1.5, 0.6)$) -- ($(TS)+(2.3, 0.6)$);
\node[text=white, font=\scriptsize] at ($(TS)+(1.9, 0.85)$) {$\Delta t$};

% Arrow to histogram
\draw[arr, labelcol] ($(TS)+(2.25,-0.6)$) -- ($(TS)+(2.25,-1.3)$);

% Axes
\draw[arr, axiscol] ($(HIST)+(0,-0.1)$) -- ($(HIST)+(4.5,-0.1)$);

\node[text=axiscol, font=\scriptsize, rotate=90] at ($(HIST)+(-0.3, 0.7)$) {$H(\tau)$};

    ($(HIST)+(0.0, 0.675)$)
    .. controls ($(HIST)+(1.0, 0.58)$) and ($(HIST)+(2.5, 0.52)$) ..
    ($(HIST)+(4.1, 0.50)$);
\node[text=axiscol, font=\scriptsize, left=1pt] at ($(HIST)+(4.1, 0.52)$) {Poisson};

    ($(HIST)+(0.0,-0.1)$) rectangle ($(HIST)+(0.3, 1.35)$);

\fill[histcol, opacity=0.35]
    ($(HIST)+(3.3,-0.1)$) rectangle ($(HIST)+(3.6, 0.50)$);

% τ=0 label
